const player = new FloatImgPlay({ selector: '.float-imgplay', autoplay: false }); player.init(); // Register a specific element with different options const hero = document.querySelector('#hero-image'); player.register(hero, { autoplay: true, audio: { waveform: 'sawtooth', masterVolume: 0.5, pitchShiftSemitones: 3, tempo: 132, scaleMode: 'minor' } });
const player = new FloatImgPlay({ selector: '.float-imgplay' }); player.init(); // Play a specific element player.play(document.querySelector('#my-image')); // Stop a specific element player.stop(document.querySelector('#my-image')); // Pause (alias for stop) player.pause(document.querySelector('#my-image')); // Re-analyze after image src changes player.refresh(); // Remove a single element player.unregister(document.querySelector('#my-image')); // Clean up everything player.destroy();
import { FloatImgPlay, MidiExport, MetaParser } from 'float-imgplay'; // --- MIDI Export --- const player = new FloatImgPlay({ selector: '.float-imgplay' }); player.init(); // Get score from an instance and export as MIDI const inst = player.instances.get(someElement); if (inst && inst.currentScore) { MidiExport.exportAndDownload( inst.currentScore, { bpm: 120 }, 'my-image.mid' ); } // --- Embedded Metadata (PNG tEXt / EXIF / Sidecar JSON) --- // Images with embedded imgplay metadata auto-switch engines: // - meta.midi present → MIDI Engine (parses & plays MIDI) // - meta.audio present → Audio Engine (streams mp3/wav) // - no meta → Image Engine (pixel analysis) // Sidecar JSON example: place next to image as image.jpg.imgplay.json // { "imgplay": { "midi": { "url": "/songs/track.mid" } } } // --- Security Options --- const securePlayer = new FloatImgPlay({ selector: '.float-imgplay', security: { allowedDomains: ['cdn.example.com', 'img.example.com'], maxFileSize: 5242880, // 5MB allowedMimeTypes: ['image/png', 'image/jpeg'] } }); securePlayer.init();

List of allowed hostnames maxFileSize: 10485760, // Max file size in bytes (10MB). 0 = no limit allowedMimeTypes: [ // Allowed MIME types for images 'image/png', 'image/jpeg', 'image/webp', 'image/gif' ] } }).init();
